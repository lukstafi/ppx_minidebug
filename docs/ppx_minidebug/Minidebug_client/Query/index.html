Module Minidebug_client.Query

Query layer for database access

type entry = {
  1. entry_id : int;
  2. seq_num : int;
  3. parent_id : int option;
  4. parent_seq_num : int option;
  5. depth : int;
  6. message : string;
  7. location : string option;
  8. data : string option;
  9. elapsed_start_ns : int;
  10. elapsed_end_ns : int option;
  11. is_result : bool;
  12. log_level : int;
  13. entry_type : string;
}
type run_info = {
  1. run_id : int;
  2. timestamp : string;
  3. elapsed_ns : int;
  4. command_line : string;
  5. run_name : string option;
}
type stats = {
  1. total_entries : int;
  2. total_values : int;
  3. unique_values : int;
  4. dedup_percentage : float;
  5. database_size_kb : int;
}
val get_runs : Sqlite3.db -> run_info list

Get all runs from database, ordered by run_id descending

val get_latest_run_id : Sqlite3.db -> int option

Get the ID of the most recent run

val get_entries : Sqlite3.db -> run_id:int -> ?parent_id:int -> ?max_depth:int -> unit -> entry list

Get entries for a specific run, optionally filtered by parent_id and max_depth

val get_stats : Sqlite3.db -> string -> stats

Get database statistics including deduplication metrics

val search_entries : Sqlite3.db -> run_id:int -> pattern:string -> entry list

Search entries by regex pattern matching message, location, or data
